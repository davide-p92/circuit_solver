cmake_minimum_required(VERSION 3.16)
project(CircuitGUI LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Wichtig f√ºr Q_OBJECT

add_executable(CircuitGUI
    main.cpp
    mainwindow.cpp
    mainwindow.h
    algorithms.cpp
